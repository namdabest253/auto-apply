```prisma
model JobPreference {
  id          String   @id @default(cuid())
  profileId   String   @unique
  profile     Profile  @relation(fields: [profileId], references: [id], onDelete: Cascade)
  locations   String[]
  roleTypes   String[]
  industries  String[]
  keywords    String[]
}

model QAEntry {
  id          String   @id @default(cuid())
  profileId   String
  profile     Profile  @relation(fields: [profileId], references: [id], onDelete: Cascade)
  question    String
  answer      String
  sortOrder   Int      @default(0)
}
```

2. Add server actions to `src/app/(dashboard)/profile/actions.ts` (append to existing file):
   - `saveJobPreferences(data: { locations: string[], roleTypes: string[], industries: string[], keywords: string[] })`:
     - Auth check, get/create profile for user
     - Upsert JobPreference record (profileId is unique, so upsert on profileId)
     - revalidatePath("/profile")
   - `saveQAEntry(data: { id?: string, question: string, answer: string })`:
     - Auth check, get/create profile
     - If id provided: update existing QAEntry
     - If no id: create new QAEntry with sortOrder = count of existing entries
     - revalidatePath("/profile")
   - `deleteQAEntry(id: string)`:
     - Auth check, verify entry belongs to user's profile
     - Delete QAEntry
     - revalidatePath("/profile")

3. Create `src/app/(dashboard)/profile/components/job-preferences.tsx` (client component):

   - **Industries section**: Same multi-select pattern using INDUSTRIES from constants
   - **Keywords section**: TagInput component with placeholder "Add keyword (e.g. React, machine learning)"
   - **Save button**: Calls saveJobPreferences server action with all four arrays
