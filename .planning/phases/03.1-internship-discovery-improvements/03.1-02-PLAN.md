```typescript
  platform: string;
  discover(params: SearchParams): Promise<DiscoveredJob[]>;
}
export interface DiscoveredJob {
  externalUrl: string;
  platform: string;
  title: string;
  company: string;
  location: string | null;
  datePosted: Date | null;
  descriptionHtml: string | null;
  descriptionText: string | null;
  salary: string | null;
  metadata: Record<string, unknown>;
}
```

Lever API key differences from Greenhouse (from RESEARCH.md):
- Base URL: `https://api.lever.co/v0/postings/{slug}?mode=json`
- Response is raw array `[...]` (NOT `{ jobs: [...] }`)
- Title field: `posting.text` (not `job.title`)
- URL field: `posting.hostedUrl` (not `job.absolute_url`)
- Date field: `posting.createdAt` (epoch ms, not ISO string)
- Location: `posting.categories.location`
- Description: `posting.description` (HTML) + `posting.descriptionPlain`
- Salary: `posting.salaryRange` object `{currency, interval, min, max}`
- Department: `posting.categories.team` or `posting.categories.department`
